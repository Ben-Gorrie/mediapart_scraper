    def get_all_article_urls_on_page(keywords : list, page_num : int) -> list:
    """
    Gathers all the urls of all blog posts on the current page
    :param keywords: list of keywords as strings
    :param page_num: number of page to search on
    :return: list of urls as strings
    """

        urls.append("https:" + url)
    return urls

    def extract_info_from_comments(soup : bs) -> pd.DataFrame:
    """
    Get all relevant info (content of comments, comment id, reply id if comment is a reply, and number of replies) from comments page and turn it into dataframe.
    :param soup: bs4 object representing comments page
    :return: pd.Dataframe containing all relevant info. return None if there are no comments
    """
    try:
        divs = soup.find_all("div", class_="content")
        content = []
        comment_id = []
        reply_id = []
        number_of_replies = []
        for div in divs:
            content.append(div.text)
            comment_id.append(div.parent.parent.get("data-comment-id"))
            reply_id.append(div.parent.parent.get("data-comment-pid"))
            number_of_replies.append(div.parent.parent.get("data-comment-reply-count"))
        
        data = {"comment" : content, 
                "id" : comment_id, 
                "parent_id" : reply_id, 
                "replies_count" : number_of_replies}
        
        df = pd.DataFrame(data)
        return df
    except:
        return None

    def convert_french_date_to_datetime(date : str) -> datetime:
    french_months = {"janv." : "01", "fvr." : "02", "mars" : "03", "avr." : "04", "mai" : "05", "juin" : "06", "juil." : "07", "aot" : "08", "sept." : "09", "oct." : "10", "nov." : "11", "dc." : "12"}
    date_as_list = date.split(" ")
    date_as_list[1] = french_months[date_as_list[1]]
    date_string = "-".join(date_as_list)
    res = datetime.strptime(date_string, "%d-%m-%Y")
    return res

def main():
    #Update external SQL db
    start_date_time = datetime.now().replace(microsecond=0)
    line = [source_id,trigger_date_time,start_date_time,datetime.fromtimestamp(0),"WIP",0,None,None]
    db_append_line(["source_id", "trigger_date_time", "start_date_time", "end_date_time", "status", "nb_rows", "error_message", "error_type"],"dbo.ExtractionResults",line,connection)
    try:
        schema = "Mediapart"
        table_articles = f"{schema}.Articles"
        table_comments = f"{schema}.Comments"

        #Creates schema for Data Table on dbricks delta lake
        spark.sql(f"CREATE SCHEMA IF NOT EXISTS {schema}")
        
        number_of_pages = get_number_of_pages(keywords)
        
        #Get all urls
        urls = []
        for page_number in range(1, number_of_pages + 1):
            page_urls = get_all_article_urls_on_page(keywords, page_number)
            urls += page_urls
            
        #remove duplicates
        urls = list(dict.fromkeys(urls))
        
        
        frames_articles = []
        frames_comments = []
        for url in urls:
            #Get article info
            soup = get_html_and_get_soup(url)
            df_article = extract_info_from_article(soup)


            comments_link = url + "/commentaires"
            
            #Get comments info
            
            soup = get_html_and_get_soup(comments_link)
            all_comments_soup = get_all_comments(soup)
            df_comments = extract_info_from_comments(all_comments_soup)

            #Add key and url to tables
            df_article["url"] = url
            df_article["keywords"] = " ".join(keywords)
            if df_comments is not None:
                df_comments["url"] = comments_link

            frames_articles.append(df_article)
            frames_comments.append(df_comments)
            
        

        df_articles = pd.concat(frames_articles)
        df_comments = pd.concat(frames_comments)
        
        df_articles["source_id"] = source_id
        df_articles["trigger_date_time"] = trigger_date_time
        df_articles["start_date_time"] = start_date_time
        
        df_comments["source_id"] = source_id
        df_comments["trigger_date_time"] = trigger_date_time
        df_comments["start_date_time"] = start_date_time
        
        
        #Write to articles table
        spark_df = spark.createDataFrame(df_articles)
        spark_df.write.option("mergeSchema", "true").format("delta").mode("overwrite").option("path",f"dbfs:/mnt/ekiodp/rfd/{schema}/{table_articles}/").saveAsTable(table_articles)
        
        #Write to comments table
        spark_df = spark.createDataFrame(df_comments)
        spark_df.write.option("mergeSchema", "true").format("delta").mode("overwrite").option("path",f"dbfs:/mnt/ekiodp/rfd/{schema}/{table_comments}/").saveAsTable(table_comments)
        
        
        update = [datetime.now(),"OK",df_articles.shape[0] + df_comments.shape[0],None]
        db_update("dbo.ExtractionResults",["end_date_time","status","nb_rows","error_message"],"start_date_time",[start_date_time],update,connection)
    except Exception as ex:
        print(url)
        print(f"Error: {ex}")
        
        update = [datetime.now(),"FAILED",0,str(ex), type(ex)]
        db_update("dbo.ExtractionResults",["end_date_time","status","nb_rows","error_message","error_type"],"start_date_time",[start_date_time],update,connection)
        raise ex
    
    return df_articles
